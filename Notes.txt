MuleEvent event as parameter injects the current MuleEvent, from there you get access to the message:
MuleContext context = event.getMuleContext();
MuleMessage message = event.getMessage();

    <munit:test name="new-test-suiteTest" description="MUnit Test">

        <assert-object-equals:compare-objects doc:name="Assert Object Equals" yesNo="true" expected-ref="#[['a', 'b', 'c']]">
            <assert-object-equals:path-otions ref="#[{'path-key':'path-value'}]"/>
            <assert-object-equals:strings >
                <assert-object-equals:string>a</assert-object-equals:string>
                <assert-object-equals:string>b</assert-object-equals:string>
            </assert-object-equals:strings>
         </assert-object-equals:compare-objects>
